# Generate documentation ?
option(WITH_DOCUMENTATION "Generate the user manual with Doxygen." OFF)
if (WITH_DOCUMENTATION)

  # Find Doxygen.
  find_program(DOXYGEN "doxygen")
  if (NOT DOXYGEN)
    message(FATAL_ERROR "Could not find Doxygen executable.")
  endif ()

  # Run Doxygen.
  execute_process(COMMAND "${DOXYGEN}" "Doxyfile"
    WORKING_DIRECTORY "${PROJECT_SOURCE_DIR}/build/doc")

  # Adjust autogenerated files.
  set(REFMAN_FILE "${PROJECT_SOURCE_DIR}/build/doc/latex/refman.tex")
  file(STRINGS "${REFMAN_FILE}" REFMAN_TEX)
  list(GET REFMAN_TEX 0 refman_line)
  file(WRITE "${REFMAN_FILE}" "${refman_line}\n")
  list(REMOVE_AT REFMAN_TEX 0)
  file(APPEND "${REFMAN_FILE}" "\\usepackage{pdflscape}\n")
  foreach (refman_line IN LISTS REFMAN_TEX)
    file(APPEND "${REFMAN_FILE}" "${refman_line}\n")
  endforeach()
  set(DOXYSTY_FILE "${PROJECT_SOURCE_DIR}/build/doc/latex/doxygen.sty")
  file(APPEND "${DOXYSTY_FILE}"
    "\\RequirePackage{longtable}\n"
    "\\renewenvironment{TabularC}[1]\n"
    "{\n"
    "\\setlength{\\tmplength}\n"
    "     {\\linewidth/(#1)-\\tabcolsep*2-\\arrayrulewidth*(#1+1)/(#1)}\n"
    "      \\par\\begin{longtable}\n"
    "             {*{#1}{|p{\\the\\tmplength}}|}\n"
    "}\n"
    "{\\end{longtable}\\par}\n")

  # Copy images in latex directory.
  file(COPY "${PROJECT_SOURCE_DIR}/doc/images/"
    DESTINATION "${PROJECT_SOURCE_DIR}/build/doc/latex"
    FILES_MATCHING PATTERN "*.png")

  # Add documentation target.
  add_custom_target("documentation" ALL "make"
    DEPENDS "${PROJECT_SOURCE_DIR}/build/doc/latex/refman.tex"
    WORKING_DIRECTORY "${PROJECT_SOURCE_DIR}/build/doc/latex")

endif ()
