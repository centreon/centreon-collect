# Global options.

set(INC_DIR "${PROJECT_SOURCE_DIR}/http_tsdb/inc/com/centreon/broker/http_tsdb")
set(HTTP_CLIENT_INC_DIR "${PROJECT_SOURCE_DIR}/http_client/inc")
set(SRC_DIR "src")
set(TEST_DIR "${PROJECT_SOURCE_DIR}/http_tsdb/test")
include_directories(inc ${INC_DIR} ${HTTP_CLIENT_INC_DIR})

# Sources.
set(SOURCES
    ${SRC_DIR}/factory.cc
    ${SRC_DIR}/column.cc
    ${SRC_DIR}/line_protocol_query.cc
    ${SRC_DIR}/stream.cc
)

# Headers.
set(HEADERS
    ${INC_DIR}/factory.hh
    ${INC_DIR}/stream.hh
    ${INC_DIR}/column.hh
    ${INC_DIR}/line_protocol_query.hh
)

add_library(http_tsdb STATIC ${SOURCES} ${HEADERS})
target_include_directories(http_tsdb PRIVATE ${INC_DIR})

target_precompile_headers(http_tsdb PRIVATE precomp_inc/precomp.hh)

set_target_properties(http_tsdb PROPERTIES COMPILE_FLAGS "-fPIC")

# Testing.
if(WITH_TESTING)
    set(TESTS_SOURCES
        ${TESTS_SOURCES}
        ${TEST_DIR}/factory_test.cc
        ${TEST_DIR}/stream_test.cc
        PARENT_SCOPE)
endif(WITH_TESTING)
